{`
const compareFunction = (a, b, col) => a[col - 1] - b[col - 1] || b[0] - a[0];

function solution(data, col, row_begin, row_end) {
    data.sort((a, b) => compareFunction(a, b, col))
    
    const slicedData = data.slice(row_begin - 1, row_end)
    
    return answer = slicedData
        .map((datam, i) => datam
             .map(v => v % (row_begin + i))
             .reduce((acc, col) => acc + col, 0))
        .reduce((acc, v) => acc ^ v)
}
`}
</CodeBlock>
