{`
  
const timeToMins = (time) => {
        const [hour, minutes] = time.split(":").map(Number)
        return hour * 60 + minutes;
}

function solution(book_time) {
    const timeArr = Array.from({length: timeToMins('23:59') + 10}, () => 0)
    
    book_time.forEach(time => {
        const [startTime, endTime] = time;
        let startMinutes = timeToMins(startTime);
        const endMinutes = timeToMins(endTime) + 9;
        
        for(startMinutes; startMinutes <= endMinutes; startMinutes++) {
            timeArr[startMinutes]++
        }
    })

    return Math.max(...timeArr);
}
`}
</CodeBlock>
